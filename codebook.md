# Create a blank workbook
OUT <- createWorkbook()

# Add some sheets to the workbook
addWorksheet(OUT, "Table5")
addWorksheet(OUT, "Table5a")
addWorksheet(OUT, "Table6a")

# Write the data to the sheets
writeData(OUT, sheet = "Table5", x = UDScodebook.T5)
writeData(OUT, sheet = "Table5a", x = UDScodebook.T5a)
writeData(OUT, sheet = "Table6a", x = UDScodebook.T6a)

# Reorder worksheets
#worksheetOrder(OUT) <- c(3,2,1) # no need to reorder at the moment

# Export the file
